						set_cosmetic_tag = LEAGUE_OF_NATIONS_LEADER

	player_elect_league_of_nations_member_window = {
		context_type = selected_country_context
		
		window_name = "player_elect_league_of_nations_member_window"
		parent_window_token = selected_country_view
		
		visible = {
			has_country_flag = member_of_league_of_nations
			ROOT = {
				is_ai = no
				has_country_flag = member_of_league_of_nations
			}
		}
		
		effects = {
			player_elect_league_of_nations_member_button_click = {
				if = {
					limit = {
						ROOT = {
							has_country_flag = player_selected_country_to_vote_for_before_league_of_nations_election
						}
					}
					ROOT = {
						clr_country_flag = player_selected_country_to_vote_for_before_league_of_nations_election
					}
					if = {
						limit = {
							has_country_flag = elected_by_player_league_of_nations_member_@ROOT
						}
						clr_country_flag = elected_by_player_league_of_nations_member_@ROOT
						else = {
							if = {
								limit = {
									any_other_country = {
										has_country_flag = elected_by_player_league_of_nations_member_@ROOT
									}
								}
								every_other_country = {
									limit = {
										has_country_flag = elected_by_player_league_of_nations_member_@ROOT
									}
									clr_country_flag = elected_by_player_league_of_nations_member_@ROOT
								}
							}
							set_country_flag = elected_by_player_league_of_nations_member_@ROOT
							ROOT = {
								set_country_flag = player_selected_country_to_vote_for_before_league_of_nations_election
							}
						}
					}
				}
				else_if = {
					limit = {
						ROOT = {
							NOT = { has_country_flag = player_selected_country_to_vote_for_before_league_of_nations_election }
						}
					}
					set_country_flag = elected_by_player_league_of_nations_member_@ROOT
					ROOT = {
						set_country_flag = player_selected_country_to_vote_for_before_league_of_nations_election
					}
				}
			}
		}
		
		triggers = {
			player_elect_league_of_nations_member_button_click_enabled = {
				NOT = { check_variable = { global.days_til_next_league_of_nations_election < 5 } }
			}
			player_elect_league_of_nations_member_voted_icon_visible = {
				has_country_flag = elected_by_player_league_of_nations_member_@ROOT
				ROOT = {
					has_country_flag = player_selected_country_to_vote_for_before_league_of_nations_election
				}
			}
		}
